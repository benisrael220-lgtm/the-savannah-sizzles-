    // Compose an email to the business owner with the booking details.
    function composeEmail(e){
      e.preventDefault();
      const name = document.getElementById('name').value.trim();
      const phone = document.getElementById('phone').value.trim();
      const type = document.getElementById('eventType').value || 'Not specified';
      const date = document.getElementById('date').value || 'Flexible';
      const guests = document.getElementById('guests').value || 'TBA';
      const details = document.getElementById('details').value.trim() || 'No additional details';

      const subject = encodeURIComponent('Booking Request from ' + name);
      const bodyLines = [
        `Name: ${name}`,
        `Phone: ${phone}`,
        `Event type: ${type}`,
        `Preferred date: ${date}`,
        `Estimated guests: ${guests}`,
        '',
        'Details / Requests:',
        details,
        '',
        '-- Sent from Savannah Sizzle website'
      ];
      const body = encodeURIComponent(bodyLines.join('\n'));

      // Business email fallback. If you have a dedicated email, replace below:
      const businessEmail = 'savannahsizzle@example.com'; // <-- replace with real email if available

      // mailto link
      const mailto = `mailto:${businessEmail}?subject=${subject}&body=${body}`;
      // open mail client
      window.location.href = mailto;

      // show user note
      document.getElementById('formNote').textContent = 'Your email client should open with the request. If not, call 0729 249 870 or message us on Instagram.';
    }
